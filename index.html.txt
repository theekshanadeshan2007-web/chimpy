    <script>
        // Initialize Lucide icons
        lucide.createIcons();
        
        // Modal Management
        function openModal(type) {
            const overlay = document.getElementById('modal-overlay');
            const modal = document.getElementById(type + '-modal');
            const content = modal.querySelector('.modal-content');
            
            overlay.classList.remove('hidden');
            modal.classList.remove('hidden');
            
            // Trigger animation
            setTimeout(() => {
                overlay.classList.remove('opacity-0');
                content.classList.remove('scale-95', 'opacity-0');
                content.classList.add('scale-100', 'opacity-100');
            }, 10);
        }
        
        function closeModal() {
            const overlay = document.getElementById('modal-overlay');
            const modals = document.querySelectorAll('[id$="-modal"]');
            
            overlay.classList.add('opacity-0');
            
            modals.forEach(modal => {
                if (!modal.classList.contains('hidden')) {
                    const content = modal.querySelector('.modal-content');
                    content.classList.remove('scale-100', 'opacity-100');
                    content.classList.add('scale-95', 'opacity-0');
                }
            });
            
            setTimeout(() => {
                overlay.classList.add('hidden');
                modals.forEach(modal => modal.classList.add('hidden'));
            }, 300);
        }
        
        function switchModal(type) {
            closeModal();
            setTimeout(() => openModal(type), 300);
        }
        
        function handleSubmit(e, type) {
            e.preventDefault();
            const btn = e.target.querySelector('button[type="submit"]');
            const originalText = btn.textContent;
            
            btn.textContent = type === 'signup' ? 'Creating Account...' : 'Signing In...';
            btn.disabled = true;
            
            setTimeout(() => {
                btn.textContent = type === 'signup' ? 'Account Created!' : 'Welcome Back!';
                btn.classList.remove('bg-yellow-400', 'text-black');
                btn.classList.add('bg-green-500', 'text-white');
                
                setTimeout(() => {
                    closeModal();
                    btn.textContent = originalText;
                    btn.disabled = false;
                    btn.classList.add('bg-yellow-400', 'text-black');
                    btn.classList.remove('bg-green-500', 'text-white');
                    e.target.reset();
                }, 1000);
            }, 1500);
        }
        
        // Tool Activation
        function activateTool(tool) {
            const messages = {
                acorn: "Activating Acorn Gatherer... Scanning for keyword opportunities!",
                forest: "Opening Forest View... Loading industry dashboard.",
                winter: "Initializing Winter Prep... Analyzing market predictions."
            };
            
            // Create toast notification
            const toast = document.createElement('div');
            toast.className = 'fixed bottom-8 right-8 bg-yellow-400 text-black px-6 py-4 rounded-lg font-semibold shadow-2xl transform translate-y-20 transition-transform duration-300 z-50 flex items-center space-x-3';
            toast.innerHTML = `
                <i data-lucide="check-circle" class="w-5 h-5"></i>
                <span>${messages[tool]}</span>
            `;
            document.body.appendChild(toast);
            lucide.createIcons();
            
            setTimeout(() => toast.classList.remove('translate-y-20'), 100);
            setTimeout(() => {
                toast.classList.add('translate-y-20');
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }
        
        // Intersection Observer for scroll animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };
        
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('animate-fade-in');
                }
            });
        }, observerOptions);
        
        // Observe all cards
        document.querySelectorAll('.tool-card').forEach(card => {
            observer.observe(card);
        });
        
        // Parallax effect for hero section
        window.addEventListener('scroll', () => {
            const scrolled = window.pageYOffset;
            const parallax = document.querySelector('.animate-float');
            if (parallax) {
                parallax.style.transform = `translateY(${scrolled * 0.5}px)`;
            }
        });
